       - Import prisma from './db'
       - Import Resend from 'resend'

       Configure betterAuth with:
       - database: prismaAdapter(prisma, { provider: 'postgresql' })
       - emailAndPassword: {
           enabled: true,
           requireEmailVerification: true,
           sendVerificationEmail: async ({ user, url }) => send via Resend with from 'noreply@openvideo.dev' (or use env var RESEND_FROM_EMAIL with fallback), subject "Verify your email address", simple HTML with verification link
           sendResetPassword: async ({ user, url }) => send via Resend with password reset subject and link
         }
       - socialProviders: {
           github: { clientId: process.env.GITHUB_CLIENT_ID!, clientSecret: process.env.GITHUB_CLIENT_SECRET! },
           google: { clientId: process.env.GOOGLE_CLIENT_ID!, clientSecret: process.env.GOOGLE_CLIENT_SECRET! }
         }
       - plugins: [
           organization({
             sendInvitationEmail: async ({ email, invitationLink }) => send via Resend with invitation subject and link
           })
         ]
       - accountLinking: { enabled: true, requireEmailVerification: true }
       - Export as `auth`

       Create Resend client at top of file: `const resend = new Resend(process.env.RESEND_API_KEY)`
