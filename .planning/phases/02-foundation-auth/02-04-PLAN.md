    1. Create `editor/src/lib/ratelimit.ts`:
       - Import { Ratelimit } from '@upstash/ratelimit'
       - Import { Redis } from '@upstash/redis'

       - Create Redis client:
         ```
         const redis = new Redis({
           url: process.env.UPSTASH_REDIS_REST_URL!,
           token: process.env.UPSTASH_REDIS_REST_TOKEN!,
         })
         ```

       - Create tiered rate limiters object:
         ```
         export const rateLimiters = {
           free: new Ratelimit({
             redis,
             limiter: Ratelimit.slidingWindow(10, '10 s'),
             analytics: true,
             prefix: 'rl:free',
           }),
           pro: new Ratelimit({
             redis,
             limiter: Ratelimit.slidingWindow(60, '10 s'),
             analytics: true,
             prefix: 'rl:pro',
           }),
           enterprise: new Ratelimit({
             redis,
             limiter: Ratelimit.slidingWindow(200, '10 s'),
             analytics: true,
             prefix: 'rl:enterprise',
           }),
         }
         ```

       - Export `getRateLimiter(tier: string)`:
         - Return `rateLimiters[tier as keyof typeof rateLimiters] ?? rateLimiters.free`
         - Default to 'free' tier for unknown tiers

       - Export `formatRateLimitHeaders(result: { limit: number; remaining: number; reset: number })`:
         - Returns a Headers-compatible object:
           ```
           {
             'X-RateLimit-Limit': result.limit.toString(),
             'X-RateLimit-Remaining': result.remaining.toString(),
             'X-RateLimit-Reset': result.reset.toString(),
             'Retry-After': Math.ceil((result.reset - Date.now()) / 1000).toString(),
           }
           ```

       - Export type `RateLimitTier = 'free' | 'pro' | 'enterprise'`

    2. Update `editor/.env.sample` - append:
       ```
       # Upstash Redis (Rate Limiting)
       UPSTASH_REDIS_REST_URL=""
       UPSTASH_REDIS_REST_TOKEN=""
       ```

  <action>
    1. Create `editor/src/lib/api-middleware.ts`:
       - Import { validateApiKey, type ApiKeyContext } from './api-keys'
       - Import { getRateLimiter, formatRateLimitHeaders } from './ratelimit'
       - Import { NextResponse } from 'next/server' (or use standard Response)

       - Export type `AuthenticatedHandler = (request: Request, context: ApiKeyContext) => Promise<Response>`

       - Export `withApiAuth(handler: AuthenticatedHandler)`:
         Returns an async function `(request: Request) => Promise<Response>` that:

         a. Validates API key:
            ```
            const apiKeyContext = await validateApiKey(request)
            if (!apiKeyContext) {
              return Response.json(
                { error: 'Unauthorized', message: 'Invalid or missing API key' },
                { status: 401 }
              )
            }
            ```

         b. Applies rate limiting:
            ```
            const limiter = getRateLimiter(apiKeyContext.tier)
            const result = await limiter.limit(apiKeyContext.userId)
            ```

         c. If rate limited (result.success === false):
            ```
            return Response.json(
              { error: 'Rate limit exceeded', message: 'Too many requests. Please try again later.' },
              {
                status: 429,
                headers: formatRateLimitHeaders(result),
              }
            )
            ```

         d. If OK: call handler with request and apiKeyContext, then add rate limit headers to the response:
            ```
            const response = await handler(request, apiKeyContext)
            // Clone response and add rate limit headers
            const headers = new Headers(response.headers)
            const rlHeaders = formatRateLimitHeaders(result)
            Object.entries(rlHeaders).forEach(([key, value]) => headers.set(key, value))
            return new Response(response.body, {
              status: response.status,
              statusText: response.statusText,
              headers,
            })
            ```

       - Error handling: wrap entire function in try-catch, return 500 on unexpected errors

    2. Create `editor/src/app/api/v1/health/route.ts`:
       - Import { withApiAuth } from '@/lib/api-middleware'

       - Export const GET = withApiAuth(async (request, context) => {
           return Response.json({
             status: 'ok',
             timestamp: new Date().toISOString(),
             user: {
               id: context.userId,
               organizationId: context.organizationId,
               tier: context.tier,
             },
           })
         })
