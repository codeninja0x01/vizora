**triggers/renderCompleted.js:**
- key: 'renderCompleted'
- noun: 'Render'
- display.label: 'Render Completed'
- display.description: 'Triggers when a video render finishes successfully.'

REST hook operation (type: 'hook'):

performSubscribe: async function
```javascript
const performSubscribe = async (z, bundle) => {
  const response = await z.request({
    url: `${baseUrl}/webhooks`,
    method: 'POST',
    body: {
      url: bundle.targetUrl,  // Zapier provides unique webhook URL per Zap
      enabled: true,
    },
  });
  // Return data to store for later unsubscribe
  return response.data;  // { id, url, secret, ... }
};
```

performUnsubscribe: async function
```javascript
const performUnsubscribe = async (z, bundle) => {
  // bundle.subscribeData contains the response from performSubscribe
  const webhookId = bundle.subscribeData.id;
  await z.request({
    url: `${baseUrl}/webhooks/${webhookId}`,
    method: 'DELETE',
  });
};
```

perform: function that processes incoming webhook events
```javascript
const perform = async (z, bundle) => {
  // bundle.cleanedRequest contains the webhook payload
  const event = bundle.cleanedRequest;
  // Return minimal payload per user decision
  return [{
    id: event.data?.renderId || event.renderId,
    renderId: event.data?.renderId || event.renderId,
    statusUrl: `${baseUrl}/renders/${event.data?.renderId || event.renderId}`,
  }];
};
```

performList: fallback polling for Zap testing (Zapier requires this)
```javascript
const performList = async (z, bundle) => {
  const response = await z.request({
    url: `${baseUrl}/renders`,
    params: { status: 'completed', limit: '3' },
  });
  // Map to same shape as webhook payload for consistency
  return response.data.items.map(render => ({
    id: render.id,
    renderId: render.id,
    statusUrl: `${baseUrl}/renders/${render.id}`,
  }));
};
```

