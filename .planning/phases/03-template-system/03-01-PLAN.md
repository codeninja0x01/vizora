export type MergeFieldType = 'text' | 'url' | 'number' | 'color';

// A single merge field definition
export interface MergeField {
  key: string;           // Unique key: "{elementId}_{property}" per user decision (element/property names directly)
  elementId: string;     // UUID of the source element in projectData
  elementType: string;   // Clip type: 'Text', 'Video', 'Image', 'Audio', 'Caption'
  property: string;      // Property name: 'text', 'src', 'volume', 'fontSize', etc.
  fieldType: MergeFieldType; // Validation type
  defaultValue: unknown; // Value from original project at template creation time
}

// Template category enum (mirrors Prisma enum)
export type TemplateCategory =

  | 'PRESENTATIONS'

Create editor/src/lib/template-schema.ts with these exports:

1. `buildMergeFieldZodSchema(mergeFields: MergeField[])` - Takes an array of MergeField definitions and returns a Zod object schema. Maps fieldType to Zod types:
   - 'text' -> z.string()
   - 'url' -> z.string().url()
   - 'number' -> z.number()
   - 'color' -> z.string().regex(/^#[0-9A-Fa-f]{3,8}$/)
   Each field is optional (merge data can be partial, falling back to defaults).

2. `generateMergeSchema(mergeFields: MergeField[])` - Calls buildMergeFieldZodSchema and then z.toJSONSchema() to produce a JSON Schema object for storage in Template.mergeSchema. Returns the JSON Schema as a plain object.

3. `validateMergeData(mergeFields: MergeField[], data: Record<string, unknown>)` - Validates data against the dynamic schema. Returns { success: true, data } or { success: false, errors: ZodFlatError }.

4. `inferFieldType(property: string, value: unknown): MergeFieldType` - Infers the merge field type from property name and value. Rules:
   - property is 'src' or value is string starting with 'http' or 'blob:' -> 'url'
   - property contains 'color' or 'Color' or value matches hex color pattern -> 'color'
   - typeof value === 'number' -> 'number'
   - default -> 'text'

Import z from 'zod' (already installed at 4.1.13). Use z.toJSONSchema() for JSON Schema generation (native Zod 4 feature). Import MergeField and MergeFieldType from '@/types/template'.
